fn process(items) {
  sum := 0
  for value in items {
    if value > 0 {
      sum = sum + value
    } else {
      continue
    }
  }

  while sum > 10 {
    sum = sum - 1
  }

  outcome := risky() rescue {
    case e: Error => default_value,
    case _ => nil
  }

  final := outcome ensure {
    cleanup()
  }

  final
}
