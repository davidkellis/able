fn make_iter(start, finish) {
  Iterator i32 { gen =>
    i := start
    while i < finish {
      gen.yield(i)
      i = i + 1
    }
    gen.stop()
  }
}

fn ad_hoc(xs) {
  Iterator { driver =>
    for x in xs {
      if x %% 2 == 0 {
        driver.yield(x)
      }
    }
  }
}
