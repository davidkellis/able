fn match_patterns(value) {
  value match {
    case Point { x: x_val: i32, y } if x_val > 0 => x_val,
    case Wrapper { item: Point { x, y: y_val: i64 } } => y_val,
    case [head: i32, ...rest] => head
  }
}
